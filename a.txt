
Table Users {
  Id integer 
  User_code integer [primary key]
  Full_name nvarchar
  Birthday date null
  Phone_number varchar
  Email varchar
  Citizen_identification_card char null
  Home_address nvarchar null
  Date_created datetime
  Updated_date datetime
}

Table Accounts {
  id integer 
  Account_code integer [primary key]
  Email varchar
  Password varchar
  Token_refresh text null
  Id_permissions integer
  Email_verification bool
  Login_status enum
  Display_name nvarchar
  Photo_url text
  Role enum
  Last_login datetime null
  Date_created datetime
  Updated_date datetime
}

Table Role_permissions {
  Role_id integer 
  Permission_id integer [primary key]
  Date_created datetime
  Updated_date datetime
}

Table Roles {
  Role_id integer [primary key]
  Role_name integer
  Date_created datetime
  Updated_date datetime
}

Table Permissions {
  Id integer 
  Name varchar
  Code varchar [primary key]
  Route varchar
  Date_created datetime
  Updated_date datetime
}

Table Foods {
  Id integer 
  Foods_code integer [primary key]
  Foods_name nvarchar
  Type_code varchar
  Price integer
  Describe text null
  Warehouse_entry_date date
  Warehouse_money integer

  Updated_date datetime
}

Table Foods_type {
  Id integer
  Type_name nvarchar

  Date_created datetime
  Updated_date datetime
}

  Id integer

  Item_name nvarchar
  Quantity integer
